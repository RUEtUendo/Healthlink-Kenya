──────────────────────────────────────────────────────────── \begin{frame}{The Problem: Who Gets Left Behind?} \begin{columns}[T,totalwidth=\textwidth] \column{0.55\textwidth} \textbf{\color{navy}The Existing systems are \textbf{reactive} — no predictive intelligence reaches CHWs \item No tool distinguishes \textit{distance failure} from \textit{quality-of-care failure} \end{itemize} \vspace{0.3cm} \textbf{\color{navy}The shift this study makes} \medskip \begin{tikzpicture} \node[fill=ltgray,rounded corners=3pt, text width=5.0cm,align=center, font=\small,inner sep=5pt] (a) by \textbf{three simultaneous factors}: \begin{itemize} \item \textit{What a patient has} (enabling) \item \textit{Who they are} (predisposing) \item \textit{What conditions obstruct them} (need) \end{itemize} Predictive modelling integrating all three can flag disengagement risk \textbf{before a missed visit occurs}. \end{block} \vspace{0.2cm} \begin{alertblock}{Policy gap} \small County planners rely on aggregate statistics that cannot distinguish distance failure from quality-of-care failure. \end{alertblock} \end{columns} \end{frame} %% Parsimony + maintainability $\Rightarrow$ XGBoost. \end{block} \end{columns} \end{frame} %% ──────────────────────────────────────────────────────────── %% SLIDE 8 — ALGORITHM TOURNAMENT TABLE %% ──────────────────────────────────────────────────────────── \begin{frame}{Algorithm Tournament: Final Standings} \begin{columns}[T,totalwidth=\textwidth] \column{0.55\textwidth} \small \begin{tabular}{@{}lccc@{}} \toprule \textbf{Algorithm} & \textbf{AUC} & \textbf{F1} & \textbf{Type}\\ \midrule Logistic Regression & 0.614 & 0.661 & Linear\\ Ridge Classifier & 0.535 & 0.524 & Linear\\ Decision Tree & 0.564 & 0.501 & Single\\ Random Forest & 0.593 & 0.657 & Ensemble\\ AdaBoost & 0.550 & 0.934 & Boosting\\ Gradient Boosting & 0.576 & 0.934 & Boosting\\ Ensemble (Top 3) & 0.8145 & 0.9343 & Combined\\ \rowcolor{ltblue} \textbf{XGBoost\,★} & \textbf{0.8144} & \textbf{0.9343} & \textbf{Boosting}\\ \bottomrule \end{tabular} \vspace{0.2cm} {\scriptsize Stratified 80\,:\,20 split, \texttt{random\_state=42}} \column{0.43\textwidth} \begin{block}{The linear failure} \small LR AUC = 0.614 confirms the distance-access relationship is \textbf{non-linear and threshold-structured}.
